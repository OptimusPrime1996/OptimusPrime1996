Fetch data from URL
```javascript
function getpage() {
  const url =
    "Your URL"; // Replace with your URL

  fetch(url)
    .then((response) => {
      if (!response.ok) {
        throw new Error(`HTTP error! Status: ${response.status}`);
      }
      return response.json();
    })
    .then((data) => {
      console.log(data);
      console.log(data); // Logs the web page content
    })
    .catch((error) => {
      console.error("Error fetching the web page:", error);
    });
}

```
